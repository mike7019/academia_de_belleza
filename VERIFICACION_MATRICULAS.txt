VALUES ('Test Curso', 500000, 5, 'ABIERTO', CURRENT_DATE, CURRENT_DATE + INTERVAL '30 days', NULL);

-- Insertar matrícula
INSERT INTO matricula (fecha, estado, valor_base, descuento, valor_final, estudiante_id, curso_id)
VALUES (CURRENT_DATE, 'ACTIVA', 500000, 0, 500000,
        (SELECT id FROM estudiante WHERE numero_documento = '9999999'),
        (SELECT id FROM curso WHERE nombre = 'Test Curso'));

-- Verificar cupo
SELECT COUNT(*) as matriculas_activas
FROM matricula m
WHERE m.curso_id = (SELECT id FROM curso WHERE nombre = 'Test Curso')
  AND m.estado = 'ACTIVA';
-- Esperado: 1

-- Cancelar
UPDATE matricula
SET estado = 'CANCELADA'
WHERE curso_id = (SELECT id FROM curso WHERE nombre = 'Test Curso');

-- Verificar cupo liberado
SELECT COUNT(*) as matriculas_activas
FROM matricula m
WHERE m.curso_id = (SELECT id FROM curso WHERE nombre = 'Test Curso')
  AND m.estado = 'ACTIVA';
